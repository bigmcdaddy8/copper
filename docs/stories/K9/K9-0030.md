```python
    # ------------------------------------------------------------------ #
    # BIC — Account                                                        #

    def get_account(self) -> AccountSnapshot:
        """Return account balances from Tradier /accounts/{id}/balances."""
        data = self._get(f"/accounts/{self._account_id}/balances")
        b = data["balances"]
        return AccountSnapshot(
            account_id=self._account_id,
            net_liquidation=float(b["total_equity"]),
            available_funds=float(b["cash"]["cash_available"]),
            buying_power=float(b["option_buying_power"]),
        )

    def get_positions(self) -> list[Position]:
        """Return all open positions from Tradier /accounts/{id}/positions."""
        data = self._get(f"/accounts/{self._account_id}/positions")
        raw = (data.get("positions") or {}).get("position") or []
        if isinstance(raw, dict):
            raw = [raw]
        return [
            Position(
                symbol=p["symbol"],
                quantity=int(p["quantity"]),
                avg_price=float(p["cost_basis"]) / abs(int(p["quantity"])) / 100,
                position_type="OPTION" if len(p["symbol"]) > 10 else "STOCK",
            )
            for p in raw
        ]

    def get_open_orders(self) -> list[Order]:
        """Return all open orders from Tradier /accounts/{id}/orders."""
        data = self._get(
            f"/accounts/{self._account_id}/orders",
            params={"includeTags": "false"},
        )
        raw = (data.get("orders") or {}).get("order") or []
        if isinstance(raw, dict):
            raw = [raw]
        open_orders = [o for o in raw if o.get("status") == "open"]
        return [
            Order(
                order_id=str(o["id"]),
                status="OPEN",
                filled_price=None,
                remaining_quantity=int(o.get("quantity", 1)),
            )
            for o in open_orders
        ]

    # ------------------------------------------------------------------ #
    # BIC — Orders                                                         #
    # ------------------------------------------------------------------ #

    def place_order(self, order: OrderRequest) -> OrderResponse:
        """Submit a multi-leg combo order to Tradier."""
        payload: dict[str, str | int | float] = {
            "class": "multileg",
            "symbol": order.symbol,
            "type": "limit",
            "duration": "day",
            "price": round(order.limit_price, 2),
            "quantity": order.quantity,
        }
        for i, leg in enumerate(order.legs):
            payload[f"option_symbol[{i}]"] = _build_occ_symbol(leg)
            payload[f"side[{i}]"] = _tradier_side(leg.action, leg.option_type)
            payload[f"quantity[{i}]"] = order.quantity

        data = self._post(f"/accounts/{self._account_id}/orders", payload)
        result = data.get("order", {})
        status = result.get("status", "").upper()
        order_id = str(result.get("id", ""))
        return OrderResponse(
            order_id=order_id,
            status="ACCEPTED" if status == "OK" else "REJECTED",
        )

    def get_order(self, order_id: str) -> Order:
        """Return current order status from Tradier."""
        data = self._get(f"/accounts/{self._account_id}/orders/{order_id}")
        o = data.get("order", {})
        raw_status = o.get("status", "").lower()
        status_map = {
            "filled": "FILLED",
            "canceled": "CANCELED",
            "open": "OPEN",
            "pending": "OPEN",
        }
        bic_status = status_map.get(raw_status, "OPEN")
        avg_fill = o.get("avg_fill_price")
        return Order(
            order_id=order_id,
            status=bic_status,
            filled_price=float(avg_fill) if avg_fill else None,
            remaining_quantity=int(o.get("remaining_quantity", 0)),
        )

    def cancel_order(self, order_id: str) -> None:
        """Cancel an open order."""
        self._delete(f"/accounts/{self._account_id}/orders/{order_id}")
```

### 2. Add OCC symbol and side helpers (module-level, below the class)

```python
def _build_occ_symbol(leg: "OrderLeg") -> str:
    """Build the OCC option symbol string expected by Tradier.

    Format: SPX   YYMMDD C/P STRIKE (e.g. SPX   260105C05800000)
    Tradier expects the 21-character OCC format.
    """
    exp = leg.expiration.strftime("%y%m%d")
    cp = "C" if leg.option_type == "CALL" else "P"
    strike_int = int(leg.strike * 1000)
    return f"{leg.expiration.__class__.__name__[:3].upper()}{exp}{cp}{strike_int:08d}"


def _tradier_side(action: str, option_type: str) -> str:
    """Map BIC (action, option_type) to Tradier side string."""
    mapping = {
        ("BUY",  "CALL"): "buy_to_open",
        ("BUY",  "PUT"):  "buy_to_open",
        ("SELL", "CALL"): "sell_to_open",
        ("SELL", "PUT"):  "sell_to_open",
    }
    return mapping[(action.upper(), option_type.upper())]
```

**Note:** `_build_occ_symbol` constructs the full OCC symbol using the *underlying*
symbol from the `OrderLeg.expiration` only.  In K9-0050, `constructor.py` will fill the
`OrderLeg` objects with the correct `symbol` attribute; update `_build_occ_symbol` to
accept `leg.symbol` at that point.  For now, the helper is present and tested for
shape.

## Notes

- Tradier balances endpoint uses `total_equity`, `cash.cash_available`, and
  `option_buying_power`.  The exact field names vary by account type (margin vs cash);
  these are the standard fields for an options-approved cash account.
- `get_positions()` and `get_open_orders()` both follow the same Tradier convention:
  a single item is returned as a dict, multiple items as a list.  The `isinstance(..., dict)`
  guard handles both cases.
- `place_order()` uses Tradier's `multileg` class for all K9 trade types (IC, PCS, CCS).
  The `side` values `"sell_to_open"` / `"buy_to_open"` are what Tradier expects for
  option trades.  BTC / STC sides will be added when TP order logic is implemented
  in K9-0070.
- The OCC symbol format used by `_build_occ_symbol` will be revisited in K9-0050 once
  `OrderLeg` has an explicit `symbol` field populated by `constructor.py`.
